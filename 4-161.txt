using System;

namespace ConsoleApp5
{
    class Program
    {
        static void Main(string[] args)
        {
            for (int i = 1000; i <= 9999; i++)
            {
                int i1 = (int)Math.Sqrt(i * 26);
                if (i1 * i1 == i * 26)
                {
                    Console.WriteLine(i);
                    break;
                }
            }
        }
    }
}
