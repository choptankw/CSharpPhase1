using System;

namespace ConsoleApp5
{
    class Program
    {
        static void Main(string[] args)
        {
            int n = int.Parse(Console.ReadLine());
            int prod = 1;

            while (n / 10 != 0)
            {
                prod = prod * (n % 10);
                n = n / 10;
            }
            if (n / 10 == 0)
                prod = prod * n;

            Console.WriteLine(prod);
        }
    }
}
