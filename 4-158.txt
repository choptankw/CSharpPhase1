using System;

namespace ConsoleApp5
{
    class Program
    {
        static void Main(string[] args)
        {
            decimal prod = 1;

           for (int i = 100; i <= 999; i++)
            {
                if (i % 2 != 2 && i % 3 != 3)
                {
                    prod *= i;
                }
                else
                    continue;
            }

            Console.WriteLine(prod);
        }
    }
}
