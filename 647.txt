using System;

namespace ConsoleApp5
{
    class Program
    {
        static void Main(string[] args)
        {
            string n = Console.ReadLine();
            bool t = true;

            for (int i = 0; i < n.Length / 2; i++)
            {
                if (n[i] != n[n.Length - 1 - i])
                {
                    t = false;
                    break;
                }    
            }

            Console.WriteLine(t);
        }
    }
}
