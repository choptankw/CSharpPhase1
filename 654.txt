using System.Text;

namespace ConsoleApp6
{
    class Program
    {
        static void Main(string[] args)
        {
            StringBuilder s = new StringBuilder();
            string line = Console.ReadLine();

            for (int i = 0; i < line.Length; i++)
            {
                if (line[i] == 'a')
                    continue;
                else
                    s.Append(line[i]);
            }

            string s1 = s.ToString();
            Console.WriteLine(s);
        }
    }
