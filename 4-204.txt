using System;

namespace ConsoleApp5
{
    class Program
    {
        static void Main(string[] args)
        {
            int n = int.Parse(Console.ReadLine());

            while (n / 10 != 0)
            {
                Console.Write(n % 10);
                n = n / 10;
            }
            if (n < 10)
                Console.Write(n);
        }
    }
}
