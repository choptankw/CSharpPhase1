using System;
using System.Text;

namespace ConsoleApp6
{
    class Program
    {
        static void Main(string[] args)
        {
            string s = Console.ReadLine();
            StringBuilder line = new StringBuilder();

            for (int i = 0; i < s.Length; i++)
            {
                if (s[i] == 'x')
                {
                    line.Append("yy");
                }
                else
                    line.Append(s[i]);
            }

            Console.WriteLine(line.ToString());
        }
    }
}
