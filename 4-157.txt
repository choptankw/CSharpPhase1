using System;

namespace ConsoleApp5
{
    class Program
    {
        static void Main(string[] args)
        {
            int sum = 0;
            int i = 101;

            do
            {
                if (i % 5 != 0)
                    sum += i;

                i++;
            }
            while (i <= 999);

            Console.WriteLine(sum);
        }
    }
}
