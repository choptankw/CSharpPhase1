using System;

namespace ConsoleApp5
{
    class Program
    {
        static void Main(string[] args)
        {
            int n = int.Parse(Console.ReadLine());
            int sum = 0;

            while (n / 10 != 0)
            {
                sum = sum + n % 10;
                n /= 10;
            }
            if (n / 10 == 0)
                sum += n;

            Console.Write(sum);
        }
    }
}
