using System;

namespace ConsoleApp5
{
    class Program
    {
        static void Main(string[] args)
        {
            for (int i = 9999; i >= 1000; i--)
            {
                int i1 = (int)Math.Sqrt(i * 14);
                if (i1 * i1 == i * 14)
                {
                    Console.WriteLine(i);
                    break;
                }
            }
        }
    }
}
