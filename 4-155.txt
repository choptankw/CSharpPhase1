using System;

namespace ConsoleApp5
{
    class Program
    {
        static void Main(string[] args)
        {
            int sum = 0;
            int i = 12;
            
            while (i <= 99)
            {
                if (i % 3 == 0)
                    sum += i;

                i += 3;
            }

            Console.WriteLine(sum);
        }
    }
}
