using System;

namespace ConsoleApp5
{
    class Program
    {
        static void Main(string[] args)
        {
            int a = int.Parse(Console.ReadLine());

            int prod = 1;
            int i = 1;
            while (i <= a - 3)
            {
                if (a % i == 3)
                    prod *= i;
                i++;
            }

            Console.WriteLine(prod);
        }
    }
}
