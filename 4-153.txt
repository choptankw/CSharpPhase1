using System;

namespace ConsoleApp5
{
    class Program
    {
        static void Main(string[] args)
        {
            int a = int.Parse(Console.ReadLine());

            int sum = 0;

            for (int i = 3; i <= a-2; i++)
            {
                if (a % i == 2)
                    sum += i;
            }

            Console.WriteLine(sum);
        }
    }
}
